-- On your own:
SELECT
	a.Title,
	t.name,
	g.name,
	g.GenreId,
	t.UnitPrice
FROM
	albums AS a
INNER JOIN
	tracks AS t
ON
	a.AlbumId = t.AlbumId
INNER JOIN
	genres AS g
ON 
	t.GenreId = g.GenreId
ORDER BY
	t.UnitPrice DESC
LIMIT 10;
